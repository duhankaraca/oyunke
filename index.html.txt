    <div id="ui">SKOR: <span id="score">0</span></div>
    <div id="instructions">ZIPLAMAK İÇİN EKRANA DOKUN!</div>
    <canvas id="gameCanvas"></canvas>

<script>
    const canvas = document.getElementById("gameCanvas");
    const ctx = canvas.getContext("2d");
    const scoreEl = document.getElementById("score");

    // Ekran boyutlarını mobile uyarla
    canvas.width = window.innerWidth > 480 ? 400 : window.innerWidth;
    canvas.height = window.innerHeight;

    let score = 0;
    let gameOver = false;
    let gameSpeed = 5;

    // Oyuncu nesnesi
    const player = {
        x: 50,
        y: canvas.height - 100,
        size: 30,
        color: "#00fff0",
        gravity: 1.2,
        velocity: 0,
        jumpForce: -18,
        isGrounded: false
    };

    // Engeller dizisi
    let obstacles = [];

    function spawnObstacle() {
        if (gameOver) return;
        let height = Math.random() * 40 + 20;
        obstacles.push({
            x: canvas.width,
            y: canvas.height - height - 50,
            width: 20,
            height: height,
            color: "#ff007f"
        });
        // Rastgele sürelerle yeni engel gönder
        setTimeout(spawnObstacle, Math.random() * 1500 + 1000);
    }

    // Dokunma / Tıklama Kontrolü
    function doJump() {
        if (gameOver) {
            resetGame();
            return;
        }
        if (player.isGrounded) {
            player.velocity = player.jumpForce;
            player.isGrounded = false;
        }
    }

    window.addEventListener("touchstart", (e) => { e.preventDefault(); doJump(); }, { passive: false });
    window.addEventListener("mousedown", doJump);

    function resetGame() {
        obstacles = [];
        score = 0;
        gameSpeed = 5;
        gameOver = false;
        player.y = canvas.height - 100;
        player.velocity = 0;
        scoreEl.innerText = score;
        animate();
    }

    function animate() {
        if (gameOver) {
            ctx.fillStyle = "rgba(0,0,0,0.7)";
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = "#ff007f";
            ctx.font = "30px Impact";
            ctx.fillText("OYUN BİTTİ", canvas.width/2 - 60, canvas.height/2);
            ctx.font = "20px Courier New";
            ctx.fillStyle = "#00fff0";
            ctx.fillText("Tekrar için dokun", canvas.width/2 - 90, canvas.height/2 + 40);
            return;
        }

        requestAnimationFrame(animate);
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // Arka Plan Çizgisi (Yol)
        ctx.strokeStyle = "#444";
        ctx.lineWidth = 4;
        ctx.beginPath();
        ctx.moveTo(0, canvas.height - 50);
        ctx.lineTo(canvas.width, canvas.height - 50);
        ctx.stroke();

        // Oyuncu Fiziği
        player.velocity += player.gravity;
        player.y += player.velocity;

        // Yer kontrolü
        if (player.y >= canvas.height - 50 - player.size) {
            player.y = canvas.height - 50 - player.size;
            player.velocity = 0;
            player.isGrounded = true;
        }

        // Oyuncuyu Çiz
        ctx.fillStyle = player.color;
        ctx.shadowBlur = 15;
        ctx.shadowColor = player.color;
        ctx.fillRect(player.x, player.y, player.size, player.size);

        // Engelleri Güncelle ve Çiz
        for (let i = obstacles.length - 1; i >= 0; i--) {
            let o = obstacles[i];
            o.x -= gameSpeed;

            ctx.fillStyle = o.color;
            ctx.shadowColor = o.color;
            ctx.fillRect(o.x, o.y, o.width, o.height);

            // Çarpışma Testi
            if (
                player.x < o.x + o.width &&
                player.x + player.size > o.x &&
                player.y < o.y + o.height &&
                player.y + player.size > o.y
            ) {
                gameOver = true;
            }

            // Ekrandan çıkan engelleri sil ve skor ekle
            if (o.x + o.width < 0) {
                obstacles.splice(i, 1);
                score++;
                scoreEl.innerText = score;
                if (score % 5 === 0) gameSpeed += 0.5; // Oyun hızlanır
            }
        }
        ctx.shadowBlur = 0; // Performans için gölgeyi sıfırla
    }

    // İlk başlatma
    spawnObstacle();
    animate();
</script>
